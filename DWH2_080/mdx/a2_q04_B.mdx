-- 4. For 2024, show total Data Volume (KB) by Region × Quarter.
-- Regions on rows, Q1–Q4 2024 on columns.

SELECT
  {
    Descendants(
      [dim_timemonth].[Time].[year_num].&[2024],
      [dim_timemonth].[Time].[quarter_num]
    )
  } *
  { [Measures].[Data Volume (KB)] } ON COLUMNS,
  
  NON EMPTY
    [dim_city].[Geo].[region_name].Members ON ROWS

FROM [AirQ Cube]
